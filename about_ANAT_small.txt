Поскольку нет времени всё писать, вот некторые команды которые помогут Вам ее скомпилить и установить:

depmod -a
pkg-config --variable xtlibdir xtables

make clean && make install
depmod -a
modprobe xt_ANAT  

iptables -t raw -A PREROUTING -d 11.103.133.0/24 -j ANAT --dnat
iptables -A FORWARD -s 172.20.0.0/16 -i vlan3500 -o vlan102 -j ANAT --snat

modprobe xt_ANAT -r
